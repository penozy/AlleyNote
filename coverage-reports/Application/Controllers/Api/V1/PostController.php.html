967    /**
968     * 取得使用者 IP 位址
969     */
970    private function getUserIp(Request $request): string
971    {
972        $serverParams = $request->getServerParams();
973
974        // 優先順序：X-Forwarded-For > X-Real-IP > REMOTE_ADDR
975        $ipSources = [
976            'HTTP_X_FORWARDED_FOR',
977            'HTTP_X_REAL_IP',
978            'REMOTE_ADDR',
979        ];
980
981        foreach ($ipSources as $source) {
982            if (!empty($serverParams[$source])) {
983                $ip = $serverParams[$source];
984                if ($source === 'HTTP_X_FORWARDED_FOR') {
985                    // X-Forwarded-For 可能包含多個 IP，取第一個
986                    $ip = trim(explode(',', $ip)[0]);
987                }
988
989                if (filter_var($ip, FILTER_VALIDATE_IP, FILTER_FLAG_NO_PRIV_RANGE | FILTER_FLAG_NO_RES_RANGE)) {
990                    return $ip;
991                }
992            }
